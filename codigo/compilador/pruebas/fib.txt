programa fibonacci;
variables
    int : x;

funcion int fibR(int : x);
{
	si (x <= 1) entonces {
		regresa(x);
	} sino {
        regresa(fibR(x - 1) + fibR(x - 2));
	}
	%%escribe("DESPUES");
}

funcion void imprimeSequenciaFib(int : n);
{
	escribe(0);
	desde a = 1 hasta (n - 1) hacer {
		escribe(fibR(a));
	}
}

principal() {
	lee(x);
    %%escribe(fibN(x));
	imprimeSequenciaFib(x);
}
